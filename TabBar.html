Changing colors

                        
                            /* text and icon */
                            .tabs .tab a {
                                color: #000;
                            }
                            /* text and icon on hover*/
                            .tabs .tab a:hover{
                                color: #000;
                            }
                            /* text and icon active*/
                            .tabs .tab a.active {
                                color: #000;
                            }
                            /* text and icon disabled*/
                            .tabs .tab.disabled a, .tabs .tab.disabled a:hover {
                                color: rgba(177, 177,177, 0.5);
                            }
                            /* indicator */
                            .tabs .indicator {
                                background-color: #000;
                            }
                            /* background active */
                            .tabs .tab a:focus, .tabs .tab a:focus.active {
                                background-color: rgba(78, 78, 78, 0.07);
                            }
